import fetch from "node-fetch"; const owner = "kfaizan204-max"; const repo = "Care-Pro-Uniform"; const filePath = "data.json"; export default async function handler(req, res) { const token = process.env.GITHUB_TOKEN; if (!token) { return res.status(500).json({ error: "Missing GitHub token" }); } // GitHub API endpoint for file const apiUrl = `https://api.github.com/repos/${owner}/${repo}/contents/${filePath}`; try { if (req.method === "GET") { const resp = await fetch(apiUrl, { headers: { Authorization: `token ${token}` }, }); const data = await resp.json(); const content = Buffer.from(data.content, "base64").toString("utf-8"); res.status(200).json(JSON.parse(content)); return; } if (req.method === "POST") { const body = req.body; if (!body) return res.status(400).json({ error: "Missing body" }); // get current file SHA const getResp = await fetch(apiUrl, { headers: { Authorization: `token ${token}` }, }); const current = await getResp.json(); const sha = current.sha; // encode updated content const updatedContent = Buffer.from(JSON.stringify(body, null, 2)).toString("base64"); // commit to GitHub const putResp = await fetch(apiUrl, { method: "PUT", headers: { Authorization: `token ${token}`, "Content-Type": "application/json", }, body: JSON.stringify({ message: "Update data.json via Vercel API", content: updatedContent, sha, }), }); if (!putResp.ok) { const err = await putResp.text(); throw new Error(err); } res.status(200).json({ success: true }); return; } res.status(405).json({ error: "Method not allowed" }); } catch (err) { console.error("‚ùå API error:", err); res.status(500).json({ error: "Internal Server Error", message: err.message }); } }
